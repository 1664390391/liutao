/**
      * 简单的插入排序
      * 思路：默认被标记的数据，左边的都是有序的，例如当前数据，取第一个为标记数据，左边的那个0号位置是有序的
      * 然后循环，将取得的数据进行对比，如果该数据，小于左边数组中的元素，将该元素右移，空出位置，继续比较，直到比左边数组的某一个元素大
      * 这个时候，将空位置替换成标记数据，继续下一个循环
      */
     
     public void InsertSort() {
          int out,in;
          String temp = "";//定义一个空字符串，存放标记数据
          //定义循环，顺序对比
          for(out = 1;out<nElems;out++) {
              temp = arr[out]; //从第一个开始比较，存到临时数据中，后面依次循环
              in = out -1; //从上一个（左边排完序最大的）,依次拿出来比较
              while(in >= 0 && (arr[in].compareTo(temp)>0)) {
                   arr[in+1] = arr[in]; //不符合条件，那么就将下一位挪出来
                   in--;
              }
              //如果符合条件,则将该数据放置到比他大一点数据的下一位，因为arr[in]数据刚刚好小于temp了，所以放置到下一位
              arr[in+1] = temp;
          }
     }
